# Create session:
mutation CreateSession($deltas: [DeltaInput]) {
  createSession(deltaInputs: $deltas) {
    id, 
    deltas 
    { 
      content,
      start,
      stop
    }
  }
}
# Variables:
{
  "deltas": [
    {
      "content": "Thor tester",
      "start": 0,
      "stop": 10
    },
    {
      "content": " å mutere GraphQL",
      "start": 11,
      "stop": 30
    }
  ]
}
